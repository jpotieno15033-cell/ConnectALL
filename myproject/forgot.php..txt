<?php
session_start();
$db_host = 'localhost'; $db_user = 'root'; $db_pass = ''; $db_name = 'project_db';
try {
    $pdo = new PDO("mysql:host=$db_host;dbname=$db_name;charset=utf8", $db_user, $db_pass);
    $pdo->setAttribute(PDO::ATTR_ERRMODE, PDO::ERRMODE_EXCEPTION);
} catch (PDOException $e) { die("Database connection failed."); }

$error = ''; $success = ''; $security_question = ''; $username = '';

// PHASE A: Search for the username and retrieve their security question
if ($_SERVER['REQUEST_METHOD'] === 'POST' && isset($_POST['action']) && $_POST['action'] === 'find_user') {
    $username = trim($_POST['username']);
    $stmt = $pdo->prepare("SELECT security_question FROM users WHERE username = ?");
    $stmt->execute([$username]);
    $user = $stmt->fetch(PDO::FETCH_ASSOC);
    if ($user) {
        $security_question = $user['security_question'];
        $_SESSION['reset_username'] = $username;
    } else { 
        $error = "Username not found in the system."; 
    }
}

// PHASE B: Verify the security answer and update the password
if ($_SERVER['REQUEST_METHOD'] === 'POST' && isset($_POST['action']) && $_POST['action'] === 'reset_pass') {
    $username = $_SESSION['reset_username'] ?? '';
    $answer = strtolower(trim($_POST['security_answer']));
    $new_password = trim($_POST['new_password']);

    if (!empty($username) && !empty($answer) && !empty($new_password)) {
        $stmt = $pdo->prepare("SELECT security_answer FROM users WHERE username = ?");
        $stmt->execute([$username]);
        $user = $stmt->fetch(PDO::FETCH_ASSOC);

        if ($user && $user['security_answer'] === $answer) {
            $hashed = password_hash($new_password, PASSWORD_DEFAULT);
            $pdo->prepare("UPDATE users SET password = ? WHERE username = ?")->execute([$hashed, $username]);
            $success = "Password updated securely! You can now log in on the home page.";
            unset($_SESSION['reset_username']);
        } else { 
            $error = "Incorrect security answer verification."; 
            // Retain the question view if they answer incorrectly
            $stmt_re = $pdo->prepare("SELECT security_question FROM users WHERE username = ?");
            $stmt_re->execute([$username]);
            $user_re = $stmt_re->fetch(PDO::FETCH_ASSOC);
            $security_question = $user_re['security_question'];
        }
    } else { 
        $error = "Please fill out all recovery fields."; 
    }
}

if (isset($_GET['cancel_reset'])) {
    unset($_SESSION['reset_username']);
    header("Location: index.php");
    exit;
}

            <p style="color:#536471; font-size:14px; margin-top:4px;">Securely regain access to your profile</p>
        </div>

        <?php if (!empty($error)): ?>
            <div style="background:#fde8e8; color:#e11d48; padding:12px; border-radius:12px; font-size:13.5px; text-align:center; margin-bottom:16px; border:1px solid #fca5a5;"><?php echo $error; ?></div>
        <?php endif; ?>
        
        <?php if (!empty($success)): ?>
            <div style="background:#ecfdf5; color:#059669; padding:12px; border-radius:12px; font-size:13.5px; text-align:center; margin-bottom:16px; border:1px solid #6ee7b7;"><?php echo $success; ?></div>
        <?php endif; ?>

        <?php if (empty($security_question) && empty($success)): ?>
            <!-- STEP 1: INPUT USERNAME -->
            <form method="POST" action="">
                <input type="hidden" name="action" value="find_user">
                <label style="font-weight:700; font-size:13.5px; color:#536471; display:block; margin-bottom:6px;">Your Account Username:</label>
                <input type="text" name="username" class="input-field" placeholder="Type your username..." required>
                <button type="submit" class="btn-primary" style="background:#6366f1;">Verify Username</button>
            </form>
        <?php elseif (!empty($security_question)): ?>
            <!-- STEP 2: ANSWER QUESTION & CHOOSE NEW PASSWORD -->
            <form method="POST" action="">
                <input type="hidden" name="action" value="reset_pass">
                
                <div style="background: #f8fafc; border: 1px solid #e2e8f0; padding: 14px; border-radius: 12px; margin-bottom: 14px;">
                    <span style="font-size: 11px; font-weight: 800; color: #94a3b8; display: block; tracking-wider: 0.05em; text-transform: uppercase;">Security Challenge</span>
                    <strong style="color: #334155; font-size: 14.5px; display: block; margin-top: 2px;"><?php echo htmlspecialchars($security_question); ?></strong>
                </div>

                <label style="font-weight:700; font-size:13.5px; color:#536471; display:block; margin-bottom:6px;">Your Answer:</label>
                <input type="text" name="security_answer" class="input-field" placeholder="Default is 'blue'..." required>
                
                <label style="font-weight:700; font-size:13.5px; color:#536471; display:block; margin-bottom:6px;">Choose New Password:</label>
                <input type="password" name="new_password" class="input-field" placeholder="Type a strong password..." required>
                
                <button type="submit" class="btn-primary" style="background:#10b981;">Update & Save Password</button>
            </form>
        <?php endif; ?>

        <p style="text-align: center; margin-top: 20px; font-size: 13.5px; margin-bottom: 0;">
